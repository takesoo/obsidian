```
fields @timestamp, @message

| parse @message ': [*]' as req_id

| parse @message /"officeId"=>"(?<officeId>.*?)"[,|}]/

# | parse @message /"memberId"=>"(?<memberId>.*?)"[,|}]/

# | parse @message /"language"=>"(?<language>.*?)"[,|}]/

| parse @message /"workerId"=>"(?<workerId>.*?)"[,|}]/

| parse @message /"formId"=>"(?<formId>.*?)"[,|}]/

| parse @message /"versionId"=>"(?<versionId>.*?)"[,|}]/

| parse @message /"query"=>"(?<query>.*?)\(/

| filter query like /mutation/ # filter by query or mutation

# | filter query like /Fetch Form/ # filter by query kind

# | filter officeId = 16

# | filter @message like /dccab600-d85d-428c-8c2c-eea051400891/ #filter by request_id

| filter (

# ヘルスチェックを除外

(@message not like /_status/) and

(@message not like /HealthCheckController/) and

(@message not like /Completed 200 OK/) and

  

# websocketを除外

(@message not like /Service:ws/) and

(@message not like /cable/) and

  

# sentryへの通信を除外

(@message not like /sentry/) and

  

# memory inspectを除外

(@message not like /memory_inspect/) and

(@message not like /MemoryInspectController/)

)

  

| sort @timestamp desc

| limit 20
```
